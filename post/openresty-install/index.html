yum install wget gcc git pcre-devel openssl-devel gcc-c++ -y

mkdir /data/src && cd /data/src
wget https://github.com/openresty/openresty/releases/download/v1.15.8.2/openresty-1.15.8.2.tar.gz
wget https://github.com/openssl/openssl/archive/OpenSSL_1_0_2s.tar.gz
wget https://ftp.pcre.org/pub/pcre/pcre-8.43.tar.gz
tar -vxzf pcre-8.43.tar.gz
tar -vzxf OpenSSL_1_0_2s.tar.gz
tar -vzxf openresty-1.15.8.2.tar.gz
cd openresty-1.15.8.2
 
# 1.13.6使用系统安装的prce包即可编译通过
./configure --prefix=/data/openresty \
    --with-http_stub_status_module \
    --with-http_v2_module \
    --with-http_realip_module \
    --with-openssl=/data/src/openssl-OpenSSL_1_0_2s
 
 
./configure --prefix=/data/openresty \
    --with-http_stub_status_module \
    --with-http_v2_module \
    --with-http_realip_module \
    --with-openssl=/data/src/openssl-OpenSSL_1_0_2s \
    --with-pcre=/data/src/pcre-8.43
 
gmake
gmake install

# 报错解决
# 缺少pcre-devel
the HTTP rewrite module requires the PCRE library
yum install -y pcre-devel
 
# 缺少openssl-devel
./configure: error: SSL modules require the OpenSSL library.
You can either do not enable the modules, or install the OpenSSL library
into the system, or build the OpenSSL library statically from the source
with nginx by using --with-openssl=<path> option.
yum install -y openssl-devel

# 关闭ipv6 CentOS 6 2.6内核无IPv6相关配置
# net.ipv6.conf.all.disable_ipv6 = 1
# net.ipv6.conf.default.disable_ipv6 = 1
#开启路由转发
net.ipv4.ip_forward = 1
#core文件名中添加pid作为扩展名
kernel.core_uses_pid = 1
# 开启SYN洪水攻击保护
net.ipv4.tcp_syncookies = 1
#修改消息队列长度
kernel.msgmnb = 65536
kernel.msgmax = 65536
#timewait的数量，默认180000
net.ipv4.tcp_max_tw_buckets = 180000
net.ipv4.tcp_sack = 1
net.ipv4.tcp_window_scaling = 1
# TCP读取缓冲区,单位Byte
net.ipv4.tcp_rmem = 4096  87380   4194304
# TCP发送缓冲区,单位Byte
net.ipv4.tcp_wmem = 4096  16384  4194304
net.core.wmem_default = 8388608
net.core.rmem_default = 8388608
net.core.rmem_max = 16777216
net.core.wmem_max = 16777216
#每个网络接口接收数据包的速率比内核处理这些包的速率快时，允许送到队列的数据包的最大数目
net.core.netdev_max_backlog = 262144
#限制仅仅是为了防止简单的DoS 攻击
net.ipv4.tcp_max_orphans = 3276800
#未收到客户端确认信息的连接请求的最大值
net.ipv4.tcp_max_syn_backlog = 262144
net.ipv4.tcp_timestamps = 0
#内核放弃建立连接之前发送SYNACK 包的数量
net.ipv4.tcp_synack_retries = 1
#内核放弃建立连接之前发送SYN 包的数量
net.ipv4.tcp_syn_retries = 1
#启用timewait 快速回收
#net.ipv4.tcp_tw_recycle = 1 ## 4.12内核已弃用
#开启重用。允许将TIME-WAIT sockets 重新用于新的TCP 连接
net.ipv4.tcp_tw_reuse = 1
net.ipv4.tcp_fin_timeout = 1
#当keepalive 起用的时候，TCP 发送keepalive 消息的频度。缺省是2 小时
net.ipv4.tcp_keepalive_time = 30
#允许系统打开的端口范围
net.ipv4.ip_local_port_range = 10000 65000
#系统中每个端口的最大监听队列的大小
net.core.somaxconn = 2048
#最大限度使用物理内存
vm.swappiness = 0
